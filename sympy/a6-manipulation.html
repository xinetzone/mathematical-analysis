import warnings
from sympy import *
import pydot
from IPython.display import SVG

warnings.filterwarnings('ignore')
init_printing('dot')

def display_dot(expr):
    g = pydot.graph_from_dot_data(dotprint(expr))[0]
    g.set_bgcolor('lightyellow')
    t = SVG(g.create_svg())
    return t

x, y, z = symbols('x y z')

z = x + y*x

display_dot(z)
